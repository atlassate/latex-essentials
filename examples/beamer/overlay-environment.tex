\begin{frame}{Pythagorean Theorem}

\begin{theorem}<1->[Pythagoras] 
$ a^2 + b^2 = c^2$
\end{theorem}

\begin{corollary}<6->
  \begin{itemize}
    \item<7-> If a2 + b2 = c2, then the triangle is right.
    \item<8-> If a2 + b2 > c2, then the triangle is acute.
    \item<9-> If a2 + b2 < c2, then the triangle is obtuse.
  \end{itemize}
\end{corollary}

\begin{proof}<2->
  \onslide<3->$AC^2=AD \cdot AB$

  \onslide<4->$BC^2=BD \cdot AB$

  \onslide<5->$AC^2+BC^2=AD \cdot AB+BD \cdot AB=(AD+BD) \cdot AB = AB^2$
\end{proof}

\end{frame}
