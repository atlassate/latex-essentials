\begin{frame}{一元二次方程}
  一元二次方程
  \label{eq}
  \[
    ax^2 + bx + c = 0
  \]
  的根是
  \label{sol}
  \[
    x_{1,2} = \cfrac{-b \pm \sqrt{b^2-4ac}}{2a}.
  \]
\end{frame}

\begin{frame}{韦达定理}
  根据\hyperlink{eq}{一元二次方程}的\hyperlink{sol}{根}, 我们可以得到如下关系:
  \begin{align*}
    x_1 + x_2 &= -\cfrac{b}{a}\\
    x_1x_2 &= \cfrac{c}{a}
  \end{align*}
\end{frame}
