%  Exact image positioning in Beamer
\documentclass{beamer}

% Required package
\usepackage{tikz}

\begin{document}

\begin{frame}{Center positioning}
% Create TikZ picture environment
\begin{tikzpicture}[remember picture, overlay]
\node[] at (current page.center) 
{
  \includegraphics[width=0.5\textwidth]{Ques}
};
\end{tikzpicture}
\end{frame}

\begin{frame}{Right center positioning}
% Create TikZ picture environment
\begin{tikzpicture}[remember picture, overlay]
\node[] at (current page.east) 
{
  \includegraphics[width=0.5\textwidth]{Ques}
};
\end{tikzpicture}

\end{frame}

\end{document}
