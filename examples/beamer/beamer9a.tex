% Text over an image in beamer
\documentclass{beamer}

% Theme choice
\usetheme{AnnArbor}

% Required package
\usepackage{tikz}

\begin{document}

\begin{frame}{Text over an image}

\centering
\begin{tikzpicture}
\node (image) 
{
  \includegraphics[width=0.5\textwidth]{Ques.png}
};

\node
[
  fill=teal,
  align=center,
  text=white,
  font={\huge\bfseries}
] at (image.center) {A beautiful photo!};

\end{tikzpicture}
\end{frame}

\end{document}
