\documentclass{article}

\usepackage{xcolor}
\usepackage{xkeyval}

% Declaration of options
\DeclareOptionX<textboxstyle>{fcolor}{\def\FrameColor{#1}}
\DeclareOptionX<textboxstyle>{bgcolor}{\def\BGColor{#1}}

% Setting default values
\ExecuteOptionsX<textboxstyle>{%
  fcolor=red,%
  bgcolor=blue!20
}

% Definition of the command
\newcommand*\textbox[2][]{%
  \begingroup
  \setkeys{textboxstyle}{#1}
  \fcolorbox{\FrameColor}{\BGColor}{#2}
  \endgroup
}

\begin{document}

\textbox{Lorem ipsum dolor sit amet}

\textbox[fcolor=olive!90,bgcolor=red!30]{Lorem ipsum dolor sit amet}

\textbox[bgcolor=yellow]{Lorem ipsum dolor sit amet}

\end{document}
